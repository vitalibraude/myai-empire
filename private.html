שלב 1: פתח Google Sheets → צור גיליון חדש → קרא לו "aoeua submissions"
שלב 2: לחץ Extensions → Apps Script → מחק את הקוד הקיים → הדבק את הקוד הזה:
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([
    new Date().toLocaleString('he-IL'),
    data.name || '', data.email || '', data.company || '',
    data.phone || '', data.interest || '', data.message || ''
  ]);
  return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var rows = sheet.getDataRange().getValues();
  var result = [];
  for (var i = 1; i < rows.length; i++) {
    result.push({
      date: rows[i][0], name: rows[i][1], email: rows[i][2],
      company: rows[i][3], phone: rows[i][4],
      interest: rows[i][5], message: rows[i][6]
    });
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function setupHeaders() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  sheet.getRange(1,1,1,7).setValues([['תאריך','שם','אימייל','חברה','טלפון','עניין','הודעה']]);
}
